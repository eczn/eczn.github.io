    function Ws(str){
        this.fullContent = str; 
        this.tos = str.split(''); 
        this.get = (this.tos.shift).bind(this.tos); 
    }

    return Ws
})()

    • jsfunction write($doms, word, cb, echo, speed){
    speed = speed || 1; 
    if (!cb || !echo) {
        console.error('no cb and echo')
        return ; 
    }
    var wordAcc = $doms[$doms.length-1].innerHTML || ''; 
    var wordWs = new Ws(word); 

    var timer4echo = setInterval(function(){
        echo($doms, wordAcc); 
    }, 300); 

    var timer4frame = setInterval(function(){
        var e = wordWs.get(); 

        if (e) {
            wordAcc += e; 
            $doms.forEach(function(dom){
                dom.innerHTML += e; 
                $(dom).scrollTop(dom.scrollHeight);
            });
        } else {
            clearInterval(timer4frame);
            clearInterval(timer4echo);
            echo($doms, wordAcc); 

            cb && cb(wordAcc); 
        }
    }, speed);
}
