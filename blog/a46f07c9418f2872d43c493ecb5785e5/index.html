变量、函数、调用

编程语言里函数的简要描述：

函数是变量，定义在作用域里，可以被调用。调用的过程是：创建新的作用域，并在这个作用域里声明函数的形参，最后执行函数体，求得返回值

    • jsmodule.exports = getExp; 

function getExp(chars){
    let bracket = 0, args = [];

    for (let i = 0; i < chars.length; i++){
        let char = chars[i]; 

        if (char === '('){
            bracket = bracket + 1; 
            if (bracket === 1){
                // push next 
                let nextChar = chars[i + 1]; 

                if (nextChar !== '('){
                    args.push(chars[i + 1]);
                    i = i + 1; 
                }

            } else if (bracket === 2) {
                let innerExp = chars.slice(i); 
                let { exp, len } = getExp(innerExp); 

                args.push(exp); 

                i = i + len - 1; 
            }
        } else if (char === ')'){
            bracket = bracket - 1; 
            if (bracket === 0){
                // 回归 
                return {
                    exp: $(args), 
                    len: i
                } 
            }
        } else {
            if (bracket === 1){
                args.push(char);                 
            }
        }
    }

    return {
        exp: $(args), 
        len: chars.length
    }
}

function $(args){
    return args.map(val => {
        let num = parseInt(val); 
        
        if (Number.isNaN(num)){
            return val
        } else {
            return num
        }
    }); 
}

/**
 * @description parse 
 * @param { String } text 
 */
function parse(text){
    text = text.split('\n')
        .map(line => {
            let pos = line.indexOf(';'); 
    
            if (pos !== -1){
                return line.substring(0, pos)
            } else {
                return line; 
            }
        })
        .join('\n')
        .replace(/\(/g, ' ( ')
        .replace(/\)/g, ' ) ')
        .replace(/\n/g, '')
        .replace(/\r/g, '')
        .replace(/\t/g, '')


    let chars = text.split(' ')
                    .filter(e => e);
    
    let ast_blocks = parseBlock(chars); 
    
    return ast_blocks; 
}

/**
 * @description 处理行 
 * @param { Array<String> } chars 
 * @param { Array } blocks 
 */
function parseBlock(chars, blocks = []){
    let { exp, len } = getExp(chars); 

    let nextChars = chars.slice(len + 1); 

    blocks.push(exp); 

    if (nextChars.length === 0){
        return blocks; 
    } else {
        return parseBlock(nextChars, blocks); 
    }
}

module.exports = class Scope {
    /**
     * @param { Object } vars 
     * @param { outer  } Scope 
     */
    constructor(vars = {}, outer = null){
        // 变量池 
        this.vars = vars; 

        // 指向外层作用域
        this.outer = outer; 
    }

    /**
     * @description 创建作用域
     */
    extend(){
        let new_scope = new Scope(); 

        // 新作用域应该指向现在的 
        new_scope.outer = this; 

        return new_scope; 
    }

    /**
     * @description 定义变量 
     * @param { String } var_name 
     * @param { * } val 
     */
    define(var_name, val){
        this.vars[var_name] = val; 
    }

    /**
     * @description 递归地搜索作用域链，如果找不到就报错
     * @param { String } name 
     */
    find(name){
        let { vars, outer } = this; 

        if (typeof vars[name] !== 'undefined'){
            return vars[name]; 
        } else {
            if (outer){
                return outer.find(name); 
            } else {
                let msg = `Error Var '${name}' Not Found`; 
                console.error(msg); 
                return msg; 
            }
        }
    }
}

// 这个模块在最后一小节来写，
// 仅需知道，它的功能是求值数组计算式，跟上篇一样
const calc = require('./calc')

class Closure {
    /**
     * @description 构造函数 
     * @param { Array<*> } args    形式参数 
     * @param { Array<*> } fn_body 数组计算式
     * @param { Scope }    scope   作用域
     */
    constructor(args, fn_body, scope){
        this.args    =  args; 
        this.fn_body =  fn_body; 
        this.scope   =  scope; 
    }

    /**
     * @description 函数调用 
     * @param { * }        ctx  函数运行的上下文this指向
     * @param { Array<*> } active_args  实际参数 
     */
    apply(ctx, active_args){
        let { args, fn_body, scope } = this; 

        // 函数运行前，新建作用域 
        let new_scope = scope.extend(); 

        // 在新作用域上声明形参
        active_args.forEach((var_val, idx) => {
            let var_name = args[idx]; 

            // 声明 
            new_scope.define(var_name, var_val); 
        }); 

        // 定义 this 指向 
        new_scope.define('this', ctx); 

        // 新作用域上执行函数体 并返回结果 
        // calc 函数在下面实现 
        return calc(fn_body, new_scope); 
    }
}

/**
 * @description 从表达式中构造闭包 
 * @param { Array } exp 形如 ['lambda', ['x'], ['+', 'x', 'x']]
 * @param { Scope } scope 
 */
function from(exp, scope){
    let [keyword, args, fn_body] = exp; 

    if (keyword === 'lambda'){
        return new Closure(args, fn_body, scope); 
    } else {
        console.error('不是函数'); 

        return null; 
    }
}

const Scope   = require('./Scope')
    , Closure = require('./Closure')

// 更多的基础运算符 
let table = {
    '+': (...args) => args.reduce((acc, cur) => acc + cur), 
    '-': (...args) => args.reduce((acc, cur) => acc - cur), 
    '*': (...args) => args.reduce((acc, cur) => acc * cur), 
    '>': (l, r) => l > r,
    '<': (l, r) => l < r,
    '>=': (l, r) => l >= r,
    '<=': (l, r) => l <= r
}

// 解释器运行的的全局作用域
let GLOBAL_SCOPE = new Scope(); 

// calc 是递归的，递归过程解释在文末 
module.exports = function calc(exp, scope = GLOBAL_SCOPE) {
    if (!Array.isArray(exp)) {
        let exp_type = typeof exp; 

        if (exp_type === 'number') {
            return exp; 
        } else if (exp_type === 'string') {
            // 说明是变量 
            // 应该在作用域里寻找这个变量 
            return scope.find(exp); 
        }
    } else {
        // 是数组, 按照数组计算式的结构来计算 
        let [first, ...args] = exp;

        // 说明是函数
        if (first === 'lambda'){
            // 从表达式里构造闭包 
            return Closure.from(exp, scope); 
        } else if (first === 'define') {
            // 说明是声明 
            let [var_name, var_exp] = args; 

            let var_val = calc(var_exp, scope); 

            scope.define(var_name, var_val); 

        } else if (first === '?') {
            // 说明是 if 运算 
            let [condition, true_todo, false_todo] = args

            let true_or_flase = calc(condition, scope); 

            return calc(
                (true_or_flase ? true_todo : false_todo), 
                scope
            ); 
        } else {
            // 如果有，则说明应该是基础计算式, 否则是闭包 
            let fn = table[first] || scope.find(first); 

            // 计算函数的实参
            // add((1 + 1), 2) => add(2, 2) => 4 
            // (+ (+ 1 1) 2)   => (+ 2 2)   => 4 
            args = args.map(e => calc(e, scope));

            return fn.apply(scope, args);
        }
    }
}

    calc 的 递归过程 如下：

    # 如果 exp 不是数组, 则进一步判断 exp 的类型：

    1. 如果 exp 是数字，直接返回数字
      即 calc(123, scope) 会返回 123
    2. 如果 exp 是字符串，那么需要在作用域里寻找这个变量
      即 calc(‘a’, scope) 等价于 scope.find(‘a’)

    # 如果 exp 是数组，即 exp 是一个数组计算式，则判断数组计算式的第一个元素：

    1. 如果第一个元素是 lambda，则应该将这个数组计算式转化成闭包
      即调用 calc(exp, scope) 会返回： Closure.from(exp, scope)
    2. 如果第一个元素是 define，则说明是声明变量
      接下来exp的第二个元素是变量名，第三个元素是值
      调用 scope.define 声明变量
    3. 如果第一个元素是 ?, 说明应该是 if 运算
      这时候先计算数组计算式的第二个参数的结果
      如果是 true, 计算数组计算式的第三个元素的值，并返回
      如果是 false, 计算数组计算式的第四个元素的值，并返回
    4. 以上两点不满足，说明第一个元素是一个基础运算符或者自定义的闭包
      因为基础运算符对于的函数和闭包都实现了 apply 因此可以统一使用 fn.apply 进行调用
      在调用前，应该对参数进行处理，即把 参数计算出结果出来, 即递归的调用 calc
      比如 js 调用 add( (1 + 1), 2 ) 首先得求 1 + 1 然后才 add(2, 2)
      这里做的处理也是一样的，先对参数进行最简化，然后传入调用。

    # 变量、函数、调用

    闭包，是函数的声明、函数体、函数所能访问的各种变量的集合所构成的三元组。

    声明里包括了函数的形式参数，函数被调用的时候会在其闭包上的作用域上创建一个新作用域，在里面将形参声明并绑定到实际参数上，最后执行函数体。

    函数体可以简化为数组计算式，形参的处理即创建新作用域并在上面绑定实参，最后执行函数体即可。
