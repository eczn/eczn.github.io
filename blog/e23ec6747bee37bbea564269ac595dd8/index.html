制作图片轮播组件

前些日子自己做了个轮播组件，中间过程感觉挺有趣，便写了下来。

起名 BGS.js (BackGroundSlide.js)

设想：

  1. 调用 BGS.init 初始化后返回 BGS实例
  2. 等待操作 (上一张 下一张等等)

# 架构原理

# 选用 CSS 实现

依我所见，有两种比较可行的方式可以实现图片轮播

  1. 利用 img 标签 ( absolute, left: 0% => 100%, 100% => 0 )
  2. 利用 CSS background 属性 ( backgorund-position-x )

我选择第二种方案作为实施方向。

    此外 BGS 上还期望有以下方法：

    1. init: 初始化，并传入配置, **配置的变量名在init里是 config 在BGS上是 _config **
    2. toNext: 立即切换到下一张
    3. toPre: 立即切换到上一张
    4. toThere: 立即切换到第 N 张 （带动画效果）
    5. setHere: 立即切换到第 N 张 （不带动画效果，瞬变）
    6. change: 图片切换的时候的回调，回调会接受一个参数 这个参数是新位点
    7. stop: 停止自动播放
    8. start: 开始自动播放，需要传递配置参数
    9. restart: 从 stop 后恢复

var position = []; 
// 当 init 传入图片 ['1.jpg', '2.jpg', '3.jpg', '4.jpg'] 
// position 会初始化为 [0, 1, 2, 3, 4] 
// 
// 也就是说 position-x 应该是: 
// [0, 1 * 容器宽度, 2 * 容器宽度, 3 * 容器宽度, 4 * 容器宽度] 

// 求下一张应该要怎样： 
// [-3, -2, -1, 0] => [0, 1, 2, 3]
// [0, 1, 2, 3] => [-1, 0, 1, 2] 相当于全部减一
position.nextPosition = function(){
    var i = 0;
    if (this[this.length-1] == 0){
        this.setPosition(0); 
        return this; 
    }
    for (i=0;i<this.length;i++){
        this[i]--; 
    }

    return this; 
}

// 求上一张应该要怎样： 
// [0, 1, 2, 3] => [-3, -2, -1, 0] 
// [-1, 0, 1, 2] => [0, 1, 2, 3]
position.prePosition = function(){
    var i = 0 ;
    if (this[0] == 0){
        console.log('循环'); 
        this.setPosition(this.length-1);
        return this; 
    }
    for (i=0;i<this.length;i++){
        this[i]++; 
    }
    return this; 
}

// 设置哪位为 0
// [-3, -2, -1, 0] => [0, 1, 2, 3]
position.setPosition = function(id){
    if (id < 0 || id >= this.length){
        console.warn('setPosition(id): id越界');
    } else {
        this[id] = 0;  // 3 
        var i = 0;
        for (i;i<this.length;i++){
            this[i] = i-id; 
        }
    }
    return this; 
}

// 将数组格式化 其中 $BGSContent 指的是 $(config.selector)
// 如果 position 为 [-3, -2, -1, 0] 那么格式化后得到字符串： 
// "-1500px, -1000px, -500px, 0px" 
position.parse = function(){
    var i = 0; 
    var temp = ''; 
    for (i=0;i<this.length;i++){
        temp += (this[i] * $BGSContent.width()) + 'px, '; 
    }
    return temp.substring(0, temp.length-2);
}

// 格式化的时候添加偏移 其中 $BGSContent 指的是 $(config.selector)
// 如果 position 为 [-3, -2, -1, 0] 调用 position.offset(1) 将得到字符串：
// "-1499px, -999px, -499px, 1px" 
position.offset = function(e){
    var i = 0; 
    var temp = ''; 
    for (i=0;i<this.length;i++){
        temp += (this[i] * $BGSContent.width() + e) + 'px, '; 
    }
    return temp.substring(0, temp.length-2);
}

// 返回 position 哪位是 0 
// 即返回播放到哪张图片了 
// 对于 [0, 1, 2, 3], where将返回 0 
position.where = function(){
    var i=0;
    for (i=0;i<this.length;i++){
        if (this[i] == 0){
            return i; 
        }
    }
    return -1; 
}

    var isPress = false; 
    var scrollStartX = false;   
    var judgeWidth;
    var scrollDistance = 0; 
    $BGSContent.mousedown(function(e){
        isPress = true; 
        scrollStartX = e.offsetX; 
        judgeWidth = $BGSContent.width() / 10;
        $BGSContent.removeClass('BGS-transition'); 
    }); 
    $BGSContent.mouseup(function(){
        isPress = false; 
        scrollStartX = false; 
        $BGSContent.addClass('BGS-transition'); 
        if (Math.abs(scrollDistance) > judgeWidth) {
            if (scrollDistance > 0){
                toNext(); 
            } else {
                toPre(); 
            }
        } else {
            render(position.parse()); 
        }
    }); 

    $BGSContent.mousemove(function(e){
        if (isPress){
            scrollDistance = scrollStartX - e.offsetX; 
            $BGSContent.css('background-position-x', position.offset(e.offsetX - scrollStartX));
        }
    }); 
}

            $BGSContent.addClass('BGS-transition'); 
            if (Math.abs(scrollDistance) > judgeWidth) {
                if (scrollDistance > 0){
                    toNext(); 
                } else {
                    toPre(); 
                }
            } else {
                render(position.parse()); 
            }
        }); 

        $BGSContent.mousemove(function(e){
            if (isPress){
                scrollDistance = scrollStartX - e.offsetX; 
                $BGSContent.css('background-position-x', position.offset(e.offsetX - scrollStartX));
            }
        }); 
    }

    return {
        init: init,
        toNext: toNext,
        toPre: toPre,
        toThere: toThere,
        setHere: setHere,
        change: change,
        stop: stop,
        start: start,
        restart: restart
    }
})(jQuery);
