/**
 * @description diff t1 and t2 
 * @param   { VNode  }    t1 源树  
 * @param   { VNode  }    t2 目标树 
 * @returns { TreePatch } diff 结果 
 */
function diff(t1, t2){
    let strTable = {}

    if (!t1 || !t2) return null; 
    
    let childrenDiff = list_diff(t1.children, t2.children, 'vid');

    let propDiff = {}; 
    Object.keys(t1.props).forEach(t1_prop => {
        let t1_val = t1.props[t1_prop]; 
        let t2_val = t2.props[t1_prop]; 
        if (t1_val != t2_val) {
            propDiff[t1_prop] = t2_val; 
        }
    }); 

    return {
        childrenDiff, propDiff
    }; 
}

 * @description 把 diff 结果应用到 t1 上
 * @param { VNode }   t1 
 * @param { TreePatch } Patches 
 */
diff.patch = (t1, tree_patch) => {
    let { childrenDiff, propDiff } = tree_patch; 

    let { children } = t1; 

    // prop patch 
    Object.keys(propDiff).forEach(key => {
        let newVal = propDiff[key]; 
        t1.props[key] = newVal; 
    })

    // children patch 
    childrenDiff.to(children); 
}

    主要是根据 type 进行数组操作，这样就可以完成虚拟 DOM 的 diff 操作了。

    • js/**
 * @description patch 2 dom 
 * @param { tree_patch } diffRes diff 结果
 */
VNode.prototype.$patch = function(tree_patch){
    let { childrenDiff, propDiff } = tree_patch; 
    let $ = this.$(); 

    // 属性的 Diff 
    Object.keys(propDiff).forEach(key => {
        let newVal = propDiff[key]; 
        $.setAttribute(key, newVal); 
    }); 

    // 子代 Diff
    let children  = this.children; 
    let $children = $.childNodes; 

    childrenDiff.forEach(patch => {
        let { idx, item, type } = patch;        
        let $ref = $children[idx]; 

        // INSERT, REORDER, DELETE
        if (type === DELETE) { // 删除 
            $.removeChild($ref);
        } else if (type === INSERT) { // 插入
            // console.log(item.$(), $ref); 
            $.insertBefore(
                item.$(),
                $ref
            ); 
        } else { // REORDER 交换
            let $temp = $children[item];
            let $next = $temp.nextSibling; 

            $.insertBefore($temp, $ref); 
            $.insertBefore($ref, $next); 
        }
    }); 
}

    利用 removeChild 来处理 DELETE, 利用 insertBefore 处理 INSERT。 至于交换顺序 REORDER 则使用了两次 insertBefore 来做。

    第一次 insertBefore 的时候，因为 $temp 已经在树的子代中了，插入之后，原来的那个会被移出，再插一次就可以完成交换了，因此预先知道了 $temp 的下一个节点了，此外，如果 $next 是 null，即 $temp 已经是最后一个节点的时候， $.insertBefore($ref, $next) 等价于 $.appendChild($ref)
