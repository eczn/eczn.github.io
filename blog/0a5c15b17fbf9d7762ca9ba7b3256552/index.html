    • js// To Generate Tab (in 4 spaces)
var nTab = n => '    '.repeat(n); 

// Token Process Table 
let tokenTable = {
    '{': (x, deep, next) => x + '\n' + nTab(deep) + next(),
    '}': (x, deep, next) => '\n' + nTab(deep) + x + next(),
    ',': (x, deep, next) => x + '\n' + nTab(deep) + next(), 
    ':': (x, deep, next) => x + ' ' + next(), 
    'default': (x, deep, next) => x + next(),
    get: function(key){
        return this[key] || this['default']
    }
}

// Deep Process Table
let deepTable = {
    '{': d => d + 1, 
    '}': d => d - 1, 
    'default': d => d,
    get: function(key){
        return this[key] || this['default']
    }
}

/**
 * @description JSON 字符数组格式化
 * @param {Array<String>} chars 
 * @param {Number} deep 
 */
function _parse(chars, deep = 0){
    let [x, ...xs] = chars; 

    if (!x) return ''; 

    let tokenProcess = tokenTable.get(x)
    // tokenTable[x] || tokenTable['default']; 
    let deepProcess = deepTable.get(x); 

    // Process 
    deep = deepProcess(deep); 

    return tokenProcess(x, deep, () => {
        return _parse(xs, deep); 
    }); 
}

/**
 * @description 正式进行格式化之前的预处理 (把字符串字符数组)
 * @param {String} s 
 */
var parse = s => _parse(
    s.split('')
); 

parse.deepTable = deepTable; 
parse.tokenProcess = tokenTable; 

module.exports = parse;
