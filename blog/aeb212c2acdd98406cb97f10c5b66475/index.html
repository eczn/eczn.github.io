构造 VNode 对象

接续前篇的 《虚拟 DOM 原理表述》，本篇继续探讨虚拟 DOM 的原理以及一个可行的实现。

上篇的主题：

虚拟 DOM 技术是指用 JavaScript 来表达 DOM 的数据结构，并由一定的算法来找出虚拟 DOM 变化前后的差异，最后将差异应用到真实的 DOM 上，从而达到性能优化的技术目的。

现在来谈谈虚拟 DOM 的数据结构，即要怎样用 JavaScript 表达 DOM，若读者无意于实现，本篇可以跳过不看了（本篇基本上是 Code）

module.exports = VNode; 

/**
 * @description 产生不重复的 id 
 * @returns { Number } 
 */
function uuid(){
    let id = vid.id || 1000; 

    vid.id = id + 1; 

    return id;
}

/**
 * @description VNode 构造标签节点
 * @param { String }        tag 
 * @param { Object }        props 
 * @param { Array<VNode> }  children 
 */
function VNode(tag = 'div', props = {}, children = []){
    this.tag = tag; 
    this.props = props; 
    this.nodeType = 1; 

    // 这里的意思是说，如果 props 里显式的钦点了 vid 那么就不生成 vid 
    if (props.vid){
        this.vid = props.vid;
    } else {
        this.vid = uuid(); 
    }
    
    this.children = children; 
} 

/**
 * @description VNode 为了方便使用取的别名 Alias
 * @param   { String }        $1 tag 
 * @param   { Object }        $2 props 
 * @param   { Array<VNode> }  $3 children 
 * @returns { VNode }
 */
VNode.h = ($1, $2, $3) => new VNode($1, $2, $3); 

/**
 * @description 构造文本节点 
 * @param { String } str 
 */
function Text(str){
    this.nodeType = 3; 
    this.vid = uuid();  
    this.text = str; 
}

/** 
 * @description 为了方便使用取的别名 Alias 
 * @param   { String } $ 文本 
 * @returns { Text }
 */
Text.t  = $ => new Text($); 
VNode.t = Text.t; 

// 挂到 VNode 上 
VNode.Text = Text;

const VNode = require('./VNode'); 

let { h } = VNode; 

module.exports = from$Node; 

/**
 * @description 根据 DOM 节点构造 VNode 对象
 * @param   { Node  } 节点 DOM 
 * @returns { VNode } 虚拟 DOM 
 */
function from$Node($node){
    let { nodeType } = $node; 
    if (nodeType === 3){
        // 过滤掉一些没用的文本节点 （全是空格和回车的无意义节点）
        if (!$node.nodeValue.trim()) return null; 
        else return new VNode.Text($node.nodeValue); 
    } else if (nodeType === 1) {
        let tag = $node.tagName.toLowerCase(); 
        
        let props = Array.from($node.attributes).reduce((props, attr) => {
            let { name, value } = attr; 
    
            props[name] = value; 
    
            return props; 
        }, {}); 
    
        let $children = Array.from($node.childNodes); 
    
        let children = $children.map($child => {
            if ($child.nodeType === 3 || $child.nodeType === 1){
                // 子节点上继续本函数 
                return from$Node($child); 
            }
        }).filter(e => e); 
    
        return h(tag, props, children);         
    }
}

/**
 * @description 根据 DOM 节点构造 VNode 对象
 * @param   { String  } html 
 * @returns { VNode } 虚拟 DOM 
 */
function from$Html(html){
    html = html.trim(); 
    
    let div = document.createElement('div'); 
    div.innerHTML = html; 
    div = div.children[0]; 

    return from$Node(div); 
}

    • js/**
 * @description 渲染为 DOM 树 
 * @return { Node } dom 树 
 */
VNode.prototype.render = function(){
    let { tag, props, vid, children, text } = this; 
  
    let $node = document.createElement(tag); 

    // Selector Bind  
    this.$ = $node; 

    Object.keys(props).forEach(key => {
        let val = props[key]; 
        
        try {   
            $node.setAttribute(key, val); 
        } catch (err) {}
    }); 

    $node.setAttribute('vid', vid); 
    
    children.forEach(child => {
        if (child.nodeType === 3){
            if (!child.text.trim()) return null;
        }

        // Text 类型也有 render 方法因此不会报错 
        let $child = child.render(); 
        $node.appendChild($child);
    }); 

    return $node; 
}

/**
 * @description 获得先前的 render 结果，如果没有则调用 this.render
 * @returns { Node } dom 节点
 */
VNode.prototype.$ = function(){
    if (this.$) return this.$; 
    else return this.render(); 
}

/**
 * @description Text 节点渲染
 * @returns { Element } 返回文本节点 
 */
Text.prototype.render = function(){
    let { text } = this; 

    let $node = document.createTextNode(text); 

    this.$ = $node; 

    return $node; 
}

/**
 * @description 获得先前的 render 结果，如果没有则调用 this.render
 * @returns { Element } 返回节点 
 */
Text.prototype.$ = function(){
    if (this.$) return this.$; 
    else return this.render(); 
}

    # TL;DR

    总之，需要实现一种抽象，将真实的 DOM 和虚拟 DOM 联系起来，在我的实现里包含：

    1. VNode 和 Text 类型来抽象表达 DOM 节点，构造虚拟 DOM
    2. 用 vid 区分不同的节点 （diff 和 patch 操作的时候需要用到）
    3. 从真实的 DOM 中生成虚拟 DOM
    4. 从虚拟 DOM 中生成真实的 DOM
    5. $ 字段表示前一次的 render 结果
