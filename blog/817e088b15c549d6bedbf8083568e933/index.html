function tplParser(root, config){
    Array.from(root.children).forEach(child => {
        // child 的属性 attributes 
        Array.from(child.attributes).forEach(attr => {
            if (attr.name.startsWith('@') && attr.value in config){
                let eventName = attr.name.slice(1); 

                child.addEventListener(
                    // such as "click"
                    eventName,
                    // such as () => console.log('clicking!') 
                    config[attr.value]
                );
            }
        }); 

        // Just Regard Child As A Tree's Root 
        tplParser(child, config); 
    })
}

    • jsfunction dataBinding(root, config){
    Array.from(root.childNodes).concat(
        Array.from(root.attributes)
    ).filter(
        e => e.nodeType === 3 || e.nodeType === 2
    ).forEach(node => {
        if (node.nodeType === 3){
            var matches = node.wholeText.match(exp);
            var tpl = node.wholeText
            // 渲染器
            var render = () => node.data = tpl.replace(exp, replacer.bind(config.data)); 
        } else {
            var matches = node.value.match(exp); 
            var tpl = node.value; 
            // 渲染器
            var render = () => node.value = tpl.replace(exp, replacer.bind(config.data)); 
        }
        
        // 无绑定 结束函数 
        if (!matches) return; 

        //  首次渲染
        render();
        
        //  根据 matches 里的匹配对象进行 setter getter 绑定 
        matches.map(e => e.slice(2, -2).trim()).forEach(varName => {
            // 原先值 
            var preVal = config.data[varName]; 
            Object.defineProperty(config.data, varName, {
                get(){
                    return preVal; 
                }, 
                set(newVal){
                    preVal = newVal; 
                    render(); 
                    return newVal; 
                }
            });
        }); 
    })
}
