活用原生方法解决数组问题

这几个月以来，我已经渐渐不用 for 来处理数组问题了。 原因主要还是 for 容易造成心智负担，而利用原生方法的 forEach 则可以让我不那么关注遍历过程，而把精力放在处理数组元素上，顺便偷偷懒 ~~

这种方法里面最最常用的是 forEach map filter reduce

而使用这些方法 函数式 地处理数组问题，会有很大的收获。

    • jsvar map = (cb, arr) => {
    // 新数组 
    let newArr = []; 
    for (let i = 0; i < arr.length; i++){
        // 这里意味着赋值 
        newArr[i] = cb(arr[i], i, arr); 
    }

    return newArr; 
}
// 以上是实现 下面是调用

}, a);

    • jsvar filter = (cb, arr) => {
    let res = []; 
    for (let i = 0; i < arr.length; i++){
        if (!!cb(arr[i], i, arr) === true){
            // 如果返回值是 true 
            // 则 push 进一个数组 
            res.push(arr[i]); 
        }
    }

    return res; 
}
// 以下是一次调用 filter 的例子 

// 我只要大于等于3的值们 
var a = [1, 2, 3, 4, 5]; 
var res = filter((item, index, itself) => {
    // 如果确实大于等于3 则返回true 即保留它 
    // 否则，返回fasle 将不会保留 
    return item >= 3; 
}, a); 

console.log(res); 
// => 
// [3, 4, 5]

    • jsvar reduce = (cb, arr, first) => {
    // acc 是 accmulate(积累) 的缩写 
    let i, acc; 
    // 如果存在 first 作为初始值
    if (first){ 
        acc = first; 
    } else { // 如果不存在 first 作为初始值
        acc = arr[0]; 
        // 从 1 开始 reduce 
        i = 1; 
    }

    for (; i < arr.length; i++){
        // 每次 cb 的返回值 作为下一次的 acc 
        acc = cb(acc, arr[i], i, arr); 
    }

    return acc; 
}
// 以下是调用 数组求和 
var a = [5, 6, 7]; 
// cur 代表 current 当前位置 
var sum = reduce((acc, cur, idx, its) => {
    return acc + cur; 
}, a, 0); // 指定第一次的 acc 是 0 

console.log(sum); 
// => 
// 18

var mapByReduce = (cb, arr) => {
    return arr.reduce((acc, cur, idx, its) => {
        let newVal = cb(cur, idx, its); 
        acc.push(newVal); 

        return acc; 
    }, []); // 指定初始acc为空数组
}

// [2, 4, 6, 8]

    # 换个角度思考数组

    我着重讨论了两个很重要的方法： map 和 filter，它们都封装了 for 的过程，因此在调用的时候可以完成这样的两件事：

    1. 映射
    2. 筛选

    如果把上面的过程看成是在数组上的 运算 ，这样来看数组具有一定的 集合性质

    此外，大部分的数组问题，其实都可以归结为若干个上述的 子运算 的问题，利用函数把他们组装起来就可以得到整个问题的解。
