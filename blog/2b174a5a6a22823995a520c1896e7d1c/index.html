    • tsximport * as React from "react";

export function Counter() {
    const [count, setCount] = React.useState(0);

    let temp1: [number, (n: number) => void];
    let temp2: [number, (n: number) => void];

    if (count >= 3) {
        temp1 = React.useState(0);
        temp2 = React.useState(6);
    } else {
        temp2 = React.useState(6);
        temp1 = React.useState(0);
    }

    return (
        <div>
            <div onClick={() => {
                setCount(count + 1);
            }}>Click Me: { count }</div>
            
            {/* test1 counter */}
            <div onClick={() => {
                temp1[1](temp1[0] + 1);
            }}>test1: { temp1[0] }</div>

            {/* test2 counter */}
            <div onClick={() => {
                temp2[1](temp2[0] + 1);
            }}>test2: { temp2[0] }</div>
        </div>
    );
}

    • tsximport * as React from "react";

type MsgCtx = {
    msg: string;
    setMsg: (msg: string) => void
}

const msgCtx = React.createContext({} as MsgCtx);

export function App() {
    return (
        <MsgCtxProvider>
            <MyInnerComponent />
        </MsgCtxProvider>
    )
}

function MsgCtxProvider(props: React.PropsWithChildren<{}>) {
    const [msg, setMsg] = React.useState('');

    const inner = (
        <div>
            <div>{ msg }</div>
            { props.children }
        </div>
    );

    return (
        <msgCtx.Provider value={{ msg, setMsg }}>
            { inner }
        </msgCtx.Provider>
    );
}

function MyInnerComponent() {
    const ctx = React.useContext(msgCtx);

    return (
        <div onClick={ () => {
            ctx.setMsg('hello, msg');
        } }>
            click here set msg:
        </div>
    )
}

    Context 的意义在于将另外一个组件里的 hooks 给其他组件用，从而达到 lift state up 的效果（上述代码中将 msg setMsg 这两个 hooks 写在了 MsgCtxProvider 里，MyInnerComponent 可以通过 useContext 来使用这两个 hooks （注：context 也可以作为依赖列表）。
